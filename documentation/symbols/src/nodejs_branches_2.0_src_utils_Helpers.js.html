  6 /**
  7  * Emit event only if there are registered listeners for the event
  8  * @param obj
  9  * @param successEvent
 10  * @param arg1
 11  * @param arg2
 12  * @private
 13  */
 14 function _emitSafeEvent(obj, successEvent, arg1, arg2, arg3) {
 15   if (obj.listeners(successEvent).length > 0) {
 16     if (typeof arg1 !== 'undefined' && typeof arg2 !== 'undefined' && typeof arg3 !== 'undefined') {
 17       obj.emit(successEvent, arg1, arg2, arg3);
 18     } else {
 19       if (typeof arg1 !== 'undefined' && typeof arg2 !== 'undefined') {
 20         obj.emit(successEvent, arg1, arg2);
 21       } else {
 22         if (typeof arg1 !== 'undefined') {
 23           obj.emit(successEvent, arg1);
 24         } else {
 25           obj.emit(successEvent);
 26         }
 27       }
 28     }
 29   }
 30 }

 32 /**
 33  * Emit ERROR event or success event
 34  * @param obj
 35  * @param err
 36  * @param errorEvent
 37  * @param successEvent
 38  * @private
 39  */
 40 exports._emitEvent = function (obj, err, errorEvent, successEvent, arg1, arg2, arg3) {
 41   if (typeof err !== 'undefined' && err !== null) {
 42     if (obj.listeners(errorEvent).length > 0) {
 43       obj.emit(errorEvent, err);
 44     }
 45   } else {
 46     _emitSafeEvent(obj, successEvent, arg1, arg2, arg3);
 47   }
 48 };

132 /**
133  * Format number values as 'money'
134  * http://stackoverflow.com/questions/149055/how-can-i-format-numbers-as-money-in-javascript/149099#149099
135  * @param decimals
136  * @param decimal_sep
137  * @param thousands_sep
138  * @return {String}
139  */
140 Number.prototype.formatAsMoney = function (decimals, decimal_sep, thousands_sep) {
141   var n = this,
142     c = isNaN(decimals) ? 2 : Math.abs(decimals), //if decimal is zero we must take it, it means user does not want to show any decimal
143     d = decimal_sep || '.', //if no decimal separator is passed we use the dot as default decimal separator (we MUST use a decimal separator)
144     t = (typeof thousands_sep === 'undefined') ? ',' : thousands_sep, //if you don't want to use a thousands separator you can pass empty string as thousands_sep value
145     sign = (n < 0) ? '-' : '',
146     i = parseInt(n = Math.abs(n).toFixed(c)) + '',
147     j;
148 
149   j = ((j = i.length) > 3) ? j % 3 : 0;
150 
151   return sign + (j ? i.substr(0, j) + t : '') + i.substr(j).replace(/(\d{3})(?=\d)/g, "$1" + t) + (c ? d + Math.abs(n - i).toFixed(c).slice(2) : '');
152 };

154 /**
155  * Escapes a string
156  * @param val
157  * @return {*}
158  * @private
159  */
160 var _escapeString = function (val) {
161   val = val.replace(/[\0\n\r\b\t\\'"\x1a]/g, function (s) {
162     switch (s) {
163       case "\0":
164         return "\\0";
165       case "\n":
166         return "\\n";
167       case "\r":
168         return "\\r";
169       case "\b":
170         return "\\b";
171       case "\t":
172         return "\\t";
173       case "\x1a":
174         return "\\Z";
175       case "\'":
176         return "''";
177       case "\"":
178         return '""';
179       default:
180         return "\\" + s;
181     }
182   });
183 
184   return val;
185 };

189 /**
190  * Replaces '?' with values from an array; also, it performs string escaping.
191  * @param sql
192  * @param arrValues
193  * @param arrDelimiters
194  * @return {*}
195  * @private
196  */
197 exports._sqlFormat = function (sql, arrValues, arrDelimiters) {
198   arrValues = [].concat(arrValues);
199   if (arrDelimiters.length !== 0) {
200     arrDelimiters = [].concat(arrDelimiters);
201   } else {
202     arrDelimiters = [];
203 
204     for (var i = arrValues.length; i > 0; --i) {
205       arrDelimiters.push("'");
206     }
207   }
208 
209   return sql.replace(/\?/g, function (match) {
210     if (!arrValues.length) {
211       return match;
212     }
213 
214     var val = arrValues.shift();
215     var delimiter = arrDelimiters.shift();
216 
217     if (val === undefined || val === null) {
218       return 'NULL';
219     }
220 
221     if (!isNaN(parseFloat(val)) && isFinite(val) && delimiter !== "'" && delimiter !== '"') {
222       return val + '';
223     }
224 
225     return delimiter + _escapeString(val) + delimiter;
226   });
227 };

238 /**
239  * Appends two buffers data into a new buffer
240  * @param buffer
241  * @param value
242  * @return {Buffer}
243  * @private
244  */
245 exports._combineData = function (buffer, value) {
246   var newBuffer = new Buffer(buffer.length + value.length);
247 
248   buffer.copy(newBuffer, 0);
249   if (Array.isArray(value)) {
250     for (var i = 0; i < value.length; i++) {
251       if (typeof  value[i] === 'string') {
252         newBuffer[buffer.length + i] = value[i].charCodeAt(0);
253       } else {
254         newBuffer[buffer.length + i] = value[i];
255       }
256     }
257   } else {
258     if (typeof value === 'Buffer') {
259       value.copy(newBuffer, buffer.length);
260     } else {
261       new Buffer(value).copy(newBuffer, buffer.length);
262     }
263   }
264 
265   return newBuffer;
266 };
