1 var DATA_TYPES = require('../constants/DataTypes'),
  2   Helpers = require('../utils/Helpers'),
  3   CAS = require('../constants/CASConstants');
  4 
  5 module.exports = BatchExecuteNoQueryPacket;
  6 
  7 /**
  8  * Constructor
  9  * @constructor
 10  */
 11 function BatchExecuteNoQueryPacket(options) {
 12   options = options || {};
 13 
 14   this.SQLs = options.SQLs;
 15   this.casInfo = options.casInfo;
 16   this.autoCommit = options.autoCommitMode;
 17 
 18   this.responseCode = 0;
 19   this.errorCode = 0;
 20   this.errorMsg = '';
 21 }
 22 
 23 /**
 24  * Write data
 25  * @param writer
 26  */
 27 BatchExecuteNoQueryPacket.prototype.write = function (writer) {
 28   var statementsLength = 0;
 29   for (var i = 0; i < this.SQLs.length; i++) {
 30     statementsLength += DATA_TYPES.INT_SIZEOF;
 31     statementsLength += (this.SQLs[i].length + 1);
 32   }
 33   var bufferLength = DATA_TYPES.DATA_LENGTH_SIZEOF + DATA_TYPES.CAS_INFO_SIZE +
 34     DATA_TYPES.BYTE_SIZEOF + DATA_TYPES.INT_SIZEOF + DATA_TYPES.BYTE_SIZEOF + statementsLength;
 35 
 36   writer._writeInt(bufferLength - DATA_TYPES.DATA_LENGTH_SIZEOF - DATA_TYPES.CAS_INFO_SIZE);
 37   writer._writeBytes(DATA_TYPES.CAS_INFO_SIZE, this.casInfo);
 38   writer._writeByte(CAS.CASFunctionCode.CAS_FC_EXECUTE_BATCH);
 39   writer._writeInt(DATA_TYPES.BYTE_SIZEOF);
 40   writer._writeByte(this.autoCommit ? 1 : 0); //Autocommit
 41   for (var j = 0; j < this.SQLs.length; j++) {
 42     writer._writeNullTerminatedString(this.SQLs[j]);
 43   }
 44 
 45   return writer;
 46 };
 47 
 48 /**
 49  * Read data
 50  * @param parser
 51  */
 52 BatchExecuteNoQueryPacket.prototype.parse = function (parser) {
 53   var reponseLength = parser._parseInt();
 54   this.casInfo = parser._parseBuffer(DATA_TYPES.CAS_INFO_SIZE);
 55 
 56   var responseBuffer = parser._parseBuffer(reponseLength);
 57   this.responseCode = parser._parseInt();
 58   if (this.responseCode < 0) {
 59     this.errorCode = parser._parseInt();
 60     this.errorMsg = parser._parseNullTerminatedString(responseBuffer.length - 2 * DATA_TYPES.INT_SIZEOF);
 61     if (this.errorMsg.length == 0) {
 62       this.errorMsg = Helpers._resolveErrorCode(this.errorCode);
 63     }
 64   }
 65 
 66   return this;
 67 };
